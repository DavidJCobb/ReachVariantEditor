timer.reset

Reset a timer to its initial value.

Example

--
-- Do something every ten seconds:
--
declare global.timer[0] = 10

do
   global.timer[0].set_rate(-100%)
   if global.timer[0].is_zero() then
      global.timer[0].reset()
      --
      -- ...and do the thing here
      --
   end
end
